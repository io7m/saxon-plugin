49    private static void createOutputDirectory(
50      final Transformation transform)
51      throws IOException
52    {
53      final File dir = new File(transform.getOutputDirectory());
54      if (dir.exists()) {
55        if (dir.isDirectory()) {
56          return;
57        }
58        throw new IOException("could not create directory: "
59          + transform.getOutputDirectory()
60          + " - already exists (and is not a directory)");
61      }
62  
63      final boolean ok = dir.mkdirs();
64      if (ok == false) {
65        throw new IOException("could not create directory: "
66          + transform.getOutputDirectory());
67      }
68    }

111     this.xslt_trans = this.xslt_exec.load();
112     this.xslt_trans.setDestination(this.output_serial);
113     this.xslt_trans.setMessageListener(this);
114     this.xslt_trans.setErrorListener(this);
115     this.xslt_trans.setSource(this.input_source);

117     final Properties parameters = transformation.getStylesheetParameters();
118     for (final Entry<Object, Object> e : parameters.entrySet()) {
119       final Object okey = e.getKey();
120       if ((okey instanceof String) == false) {
121         throw new IllegalArgumentException("Key "
122           + okey

126       }

128       final Object oval = e.getValue();

134           + ")");
135       }
136       final String val = (String) e.getValue();
137       this.xslt_trans.setParameter(new QName(key), new XdmAtomicValue(val));
138     }

141       final String path =

143       final String key = transformation.getOutputDirectoryParameterName();
144       this.xslt_trans.setParameter(new QName(key), new XdmAtomicValue(path));
145     }
